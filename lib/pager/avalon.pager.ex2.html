指定回调onJump(与例6对照看)

<!DOCTYPE html>
<html>
    <head>
        <title>分页栏组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css">
        <script src="../../avalon.js"></script>
        <script>
            avalon.library("oni", {
                $ready: function () {
                    avalon.log("控件已经构建完毕")
                }
            })
            require(["./component/pager/avalon.pager"], function () {
                avalon.define({
                    $id: "test",
                    $pager: {
                        totalItems: 150,
                        onJump: function (e, data) {
                            location.href = "#page=" + data.currentPage
                        }
                    }
                })
                avalon.scan()
            })
  
        </script>
    </head>
    <body ms-controller="test">
        <h1>指定回调onJump(与例6对照看)</h1>
        <oni:pager config="$pager"></oni:pager>
    </body>
</html>
