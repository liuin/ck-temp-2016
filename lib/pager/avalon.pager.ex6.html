总是显示上一页与下一页按钮(与例2对照看),并使用ms-attr-identifier重新指定组件VM的$id

<!DOCTYPE html>
<html>
    <head>
        <title>分页栏组件</title>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width">
        <link type="text/css" rel="stylesheet" href="../style/avalon.doc.css">
        <script src="../../avalon.js"></script>
        <script>
            avalon.library("oni", {
                $ready: function (vm) {
                    if (vm.$id.indexOf("pager") >= 0) {
                        vm.currentPage = 4
                        setTimeout(function () {
                            vm.currentPage = 3
                        }, 2000)
                    }
                }
            })

            require(["./component/pager/avalon.pager"], function () {
                avalon.define({
                    $id: "test",
                    $pager: {
                        alwaysShowNext: true,
                        alwaysShowPrev: true,
                        onJump: function (e, data) {
                            location.href = "#page=" + data.currentPage
                        }
                    }
                })
                avalon.scan()
            })

        </script>


    </head>
    <body ms-controller="test">
    <h1>总是显示上一页与下一页按钮(与例2对照看),并使用ms-attr-identifier重新指定组件VM的$id</h1>
    <oni:pager ms-attr-identifier="'pager'" config="$pager"></oni:pager><br/>
    
</body>
</html>
